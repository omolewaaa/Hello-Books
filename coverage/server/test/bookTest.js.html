describe('Book', () => {
it('it should not post a book when not an admin', (done) => {
        let item = {
            bookName: "The Lord of the Rings",
            Author: "J.R.R. Tolkien",
            bookStatus: "available",
            Details: "A great book to read",
            upvotes: 0,
            downvotes:0,
 
        };
        chai.request(app)
            .post('/api/v1/book/admin')
            .send(item)
            .end((err, res) => {
                res.should.have.status(403);
              done();
            
      });
     });
